    public class TcpServer {
    public static void main(String[] args) {
        ServerSocket serverSocket=null;
        Socket socket=null;
        InputStream is=null;
        ByteArrayOutputStream baos=null;

        try {
            //1.创建地址
            serverSocket=new ServerSocket(9999);
            //2.等待客户端连接过来
            socket =serverSocket.accept();
            //读取客户端的消息
            is=socket.getInputStream();
            //管道
            baos=new ByteArrayOutputStream();
            byte[] buffer=new byte[1024];
            Integer len;
            while ((len=is.read(buffer))!=-1){
                baos.write(buffer,0,len);
            }

            System.out.println(baos.toString());
        } catch (IOException e) {
            e.printStackTrace();
        }finally {
            //关闭资源
            if (baos!=null){
                try {
                    baos.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            //关闭资源
            if (is!=null){
                try {
                    is.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            //关闭资源
            if (socket!=null){
                try {
                    socket.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
            //关闭资源
            if (serverSocket!=null){
                try {
                    serverSocket.close();
                } catch (IOException e) {
                    e.printStackTrace();
                }
            }
        }
    }
}

    public class TcpServer {
    public static void main(String[] args) throws IOException {
        //1、创建 服务
        ServerSocket serverSocket=new ServerSocket(9000);
        //2、监听客户端的连接
        Socket socket=serverSocket.accept();//阻塞监听，会一直等待客户端连接
        //3、获取输入流
        InputStream is=socket.getInputStream();
        //4、文件输出
        FileOutputStream fos=new FileOutputStream((new File("receiver.jpg")));

        byte[] buffer=new byte[1024];
        Integer len;
        while ((len=is.read(buffer))!=-1){
            fos.write(buffer,0,len);
        }

        //通知客户端：我接收完毕了
        OutputStream os= socket.getOutputStream();
        os.write("我接收完毕了，你可以断开了。".getBytes());

        //关闭资源：实际开发中用try-catch-finally关闭
        fos.close();
        is.close();
        socket.close();
        serverSocket.close();

    }
}

    用途：发短信，需要知道对方 的地址

    发送端

    public class UdpClient {
    public static void main(String[] args) throws Exception{
        //1、建立一个Socket
        DatagramSocket socket=new DatagramSocket();

        //2、创建一个包
        String msg="hello!卖剩鸭";
        InetAddress localhost=InetAddress.getByName("localhost");
        Integer port =9090;
        //数据，数据的长度起始，要发送给谁
        DatagramPacket packet=new DatagramPacket(msg.getBytes(),0,msg.getBytes().length,localhost,port);

        //3、发送包
        socket.send(packet);

        //4、关闭流:实际开发用try-catch-finally关闭
        socket.close();

    }
}

    接收端

    public class UdpServer {
    public static void main(String[] args) throws IOException {
        //开放端口
        DatagramSocket socket = new DatagramSocket(9090);
        //接收数据包
        byte[] buffer = new byte[1024];
        DatagramPacket packet = new DatagramPacket(buffer, 0, buffer.length);

        socket.receive(packet);//阻塞接收

        System.out.println(packet.getAddress().getHostName());
        System.out.println(new String(packet.getData(), 0, packet.getLength()));

        //关闭流:实际开发用try-catch-finally关闭
        socket.close();
    }
}

    UDP循环发送消息

    发送端

    public class UdpSender {
    public static void main(String[] args) throws IOException {
        DatagramSocket socket=new DatagramSocket(8888);
        //准备数据,控制参读取 System.in
        BufferedReader reader=new BufferedReader((new InputStreamReader(System.in)));

        while (true){
            String data=reader.readLine();
            byte[] datas=data.getBytes();
            DatagramPacket packet=new DatagramPacket(datas,0,datas.length,new InetSocketAddress("localhost",6666));

            socket.send(packet);
            if (data.endsWith("bye")){
                break;
            }

        }
        socket.close();
    }
}

    接收端

    public class UdpReceive {
    public static void main(String[] args) throws IOException {
        DatagramSocket socket=new DatagramSocket(6666);

        while (true){
            //准备接收数据
            byte[] container=new byte[1024];
            DatagramPacket packet=new DatagramPacket(container,0,container.length);
            socket.receive(packet);//阻塞式接收包裹

            //断开连接
            byte[] data=packet.getData();
            String receiveData=new String(data,0,data.length);

            System.out.println(receiveData);

            if (receiveData.equals("bye")){
                break;
            }
        }
    }
}

    public class UdpTalkSendThread implements Runnable {
    DatagramSocket socket = null;
    BufferedReader reader = null;
    private Integer fromPort;
    private String toIp;
    private Integer toPort;

    public UdpTalkSendThread(Integer fromPort, String toIp, Integer toPort) {
        this.fromPort = fromPort;
        this.toIp = toIp;
        this.toPort = toPort;
        try {
            socket = new DatagramSocket(fromPort);
            reader = new BufferedReader(new InputStreamReader(System.in));
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    @Override
    public void run() {
        while (true) {
            try {
                String data = reader.readLine();
                byte[] datas = data.getBytes();
                DatagramPacket packet = new DatagramPacket(datas, 0, datas.length, new InetSocketAddress(toIp, toPort));

                socket.send(packet);
                if (data.endsWith("bye")) {
                    break;
                }
            } catch (Exception e) {
                e.printStackTrace();
            }
        }
        socket.close();

    }
}

    接收线程

    public class UdpTalkReceiveThread implements Runnable {
    DatagramSocket socket = null;
    private Integer port;
    private String msgFrom;

    public UdpTalkReceiveThread(Integer port, String msgFrom) {
        this.port = port;
        this.msgFrom = msgFrom;
        try {
            socket = new DatagramSocket(port);
        } catch (SocketException e) {
            e.printStackTrace();
        }
    }

    @Override
    public void run() {
        while (true) {
            try {
                //准备接收数据
                byte[] container = new byte[1024];
                DatagramPacket packet = new DatagramPacket(container, 0, container.length);
                socket.receive(packet);//阻塞式接收包裹

                byte[] data = packet.getData();
                String receiveData = new String(data, 0, data.length);

                System.out.println(msgFrom+":"+receiveData);

                //断开连接
                if (receiveData.equals("bye")) {
                    break;
                }
            } catch (IOException e) {
                e.printStackTrace();
            }

        }
     socket.close();
    }
}

    public class URLDownload {
    public static void main(String[] args) throws IOException {
        //1.下载地址
        String urlAddress="http://okzy.xzokzyzy.com/20200528/10692_4768c021/%E5%A2%93%E7%8E%8B%E4%B9%8B%E7%8E%8B%E7%AC%AC23%E8%AF%9D.mp4";
        URL url = new URL(urlAddress);

        //2、连接到这个资源 HTTP
        HttpURLConnection urlConnection= (HttpURLConnection) url.openConnection();

        InputStream is=urlConnection.getInputStream();

        FileOutputStream fos=new FileOutputStream("download.mp4");
        byte[] buffer=new byte[1024];
        Integer len;
        while ((len=is.read(buffer))!=-1){
            fos.write(buffer,0,len);
        }

        fos.close();
        is.close();
        urlConnection.disconnect();
    }
}
